Tìm hiểu các method của document

getElementById
document.getElementById('nameId')
getElementsByClassName
document.getElementsByClassName('nameClass')[0]
document.getElementsByClassName('nameClass')[1]
document.getElementsByClassName('nameClass')[2]
getElementsByTagName

document.getElementsByTagName('p')[0]

document.getElementsByTagName('p')[1]

document.getElementsByTagName('p')[2]

querySelector
document.querySelector('#nameIdSelector') // name Id: #name
document.querySelector('.nameClassSelector') // name Class : .name
querySelectorAll
document.querySelectorAll('#nameIdSelectorAll')
document.querySelectorAll('#nameIdSelectorAll')
